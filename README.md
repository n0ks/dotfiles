### Setup

> scripts are under development 🛑

```bash 
 git clone https://github.com/n0ks/dotfiles ~/.dotfiles 

 # mac
 cd ~/.dotfiles/mac
 chmod +x mac-setup.sh
 ./mac-setup.sh 

# source mac settings
source macos-settings 

# - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

 # linux
 cd ~/.dotfiles/linux
 chmod +x linux-deb-setup.sh
 ./linux-deb-setup.sh

```
